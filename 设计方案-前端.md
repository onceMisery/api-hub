### 6.6 API 请求层

```typescript
// src/services/request.ts
import axios, { AxiosError } from 'axios';
import { message } from 'antd';
import { useAuthStore } from '@/stores/authStore';

const request = axios.create({
  baseURL: '/api',
  timeout: 30000,
});

request.interceptors.request.use((config) => {
  const token = useAuthStore.getState().token;
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

request.interceptors.response.use(
  (response) => {
    const data = response.data;
    if (data.code !== 0) {
      message.error(data.msg || '请求失败');
      return Promise.reject(new Error(data.msg));
    }
    return data;
  },
  (error: AxiosError) => {
    if (error.response?.status === 401) {
      useAuthStore.getState().logout();
      window.location.href = '/login';
    } else if (error.response?.status === 403) {
      message.error('无权限');
    } else {
      message.error('网络异常');
    }
    return Promise.reject(error);
  }
);

export default request;
```

  token: string | null;
  refreshToken: string | null;
  user: {
    id: number;
    username: string;

    avatarUrl?: string;
  } | null;

  logout: () => void;
}

  persist(
    (set) => ({
      token: null,
      refreshToken: null,
      user: null,
      setAuth: (data) =>
        set({
          token: data.token,
          refreshToken: data.refreshToken,
          user: data.user,
        }),
      logout: () => set({ token: null, refreshToken: null, user: null }),
    }),
    { name: 'apihub-auth' }
  )
);

// 根据参数定义生成示例JSON
function generateExample(params: any[]): any {
  if (!params?.length) return {};
  const result: Record<string, any> = {};
  for (const p of params) {
    if (p.example) {
      result[p.name] = tryParseValue(p.example, p.type);
    } else if (p.children?.length) {
      if (p.type === 'array') {
        result[p.name] = [generateExample(p.children)];
      } else {
        result[p.name] = generateExample(p.children);
      }
    } else {
      result[p.name] = getDefaultByType(p.type, p.name);
    }
  }
  return result;
}

function tryParseValue(val: string, type: string): any {
  if (type === 'integer' || type === 'long') return parseInt(val);
  if (type === 'number' || type === 'double') return parseFloat(val);
  if (type === 'boolean') return val === 'true';
  return val;
}

function getDefaultByType(type: string, name: string): any {
  switch (type?.toLowerCase()) {
    case 'integer': case 'int': case 'long': return 0;
    case 'number': case 'double': case 'float': return 0.0;
    case 'boolean': return false;
    case 'array': return [];
    case 'object': return {};
    default: return `${name}_example`;
  }
}

  const [method, setMethod] = useState('GET');
  const [url, setUrl] = useState('');
  const [headers, setHeaders] = useState<{key: string; value: string}[]>([

  ]);

    enabled: !!envId,
  });

    if (currentEnv?.data?.globalHeaders) {

      const newHeaders = [...headers];

      Object.entries(globalHeaders).forEach(([key, value]) => {
        if (!existingKeys.has(key.toLowerCase())) {
          newHeaders.push({ key, value: String(value) });
        }
      });

      setHeaders(newHeaders);
    }

    try {
      const result = await debugApi.send({

        headers: Object.fromEntries(headers.filter(h => h.key).map(h => [h.key, h.value])),
        body,

    headers.forEach(h => {

    });
    if (body && ['POST', 'PUT', 'PATCH'].includes(method)) {
      cmd += ` \\\n  -d '${body}'`;

        size="small"
        items={[
          {
            key: 'headers',
            label: `Headers (${headers.length})`,
            children: (
              <HeaderEditor headers={headers} onChange={setHeaders} />
            ),
          },
          {
            key: 'body',

          },
